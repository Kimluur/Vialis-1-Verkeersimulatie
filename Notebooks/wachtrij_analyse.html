def get_perHour(df):
    multi = 36000

    perHour = df.groupby(np.arange(len(df))//multi).mean()

    every_index = []
    [every_index.append(df.iloc[index*multi]['time']) for index, row in perHour.iterrows()]

    perHour['time'] = every_index
    perHour = perHour.set_index('time')
    
    return perHour
